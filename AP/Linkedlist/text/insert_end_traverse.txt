	(using whole linked list traversal)

=====================================================================================================================

#include<conio.h>
#include<stdio.h>
#include<stdlib.h>
struct student {
	int r_no;
	float marks;
	struct student* next;
};

struct student *insert_end(int r_no , float marks , struct student *first){
	
	struct student *new;
	struct student *temp = (struct student*)malloc(sizeof(struct student));
	temp -> r_no = r_no;
	temp -> marks= marks;
	temp -> next = NULL;
	if(first == NULL){
		first = temp; 
	}
	else{
		new = first;
		while (new -> next != NULL){
			new = new -> next;
		}
		new -> next = temp;
	}
	return first;
}

void display(struct student *first){
	struct student *temp = first ;
	printf("The linked list is: \n\n");
	while(temp != NULL){
		printf("%d\t",temp->r_no);
		printf("%f",temp->marks);
		temp = temp -> next;
		printf("\t");
	}
	printf("\n");
}
int main(){
	int r_no,marks,n,i=0,no=1;
	struct student* first = NULL;

			}
		}
		else if(no == 2){
			display(temp);

		}		
	}
}
