#include<stdio.h>
#include<conio.h>
#include<string.h>

void str_to_lwr(char[]);
void main()
{
	char str[30];
	printf("Enter string: ");
	gets(str);
	str_to_lwr(str);
	printf("String after coverting to lowercase: ");
	puts(str);
	getch();
}

void str_to_lwr(char str[])
{
	char *sptr;
	int i;
	sptr=str;
	for(i=0;*(sptr+i) != '\0';i++)
	{
		if(*(sptr+i)> 64 && *(sptr+i) <92)
		{
			*(sptr+i)+=32;
		}
	}
}
