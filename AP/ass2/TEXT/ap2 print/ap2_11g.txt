#include<stdio.h>
#include<conio.h>
#include<string.h>

void copy(char[],char[]);
void main()
{
	char str1[30],str2[30];

	printf("Enter First string : ");
	gets(str1);
	printf("Enter second string to copy: ");
	gets(str2);

	copy(str1,str2);
	printf("New first string: ");
	puts(str1);
	getch();
}

void copy(char str1[],char str2[])
{
	int i;
	char *sptr1,*sptr2;
	sptr1=str1;
	sptr2=str2;

	for(i=0;*(sptr2+i)!='\0';i++)
	{
		*(sptr1+i)=*(sptr2+i);
	}
	*(sptr1+i)='\0';
}
