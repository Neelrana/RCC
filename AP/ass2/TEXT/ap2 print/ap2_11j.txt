#include<stdio.h>
#include<conio.h>
#include<string.h>

void str_to_upr(char[]);
void main()
{
	char str[30];
	printf("Enter string: ");
	gets(str);
	str_to_upr(str);
	printf("String after coverting to uppercase: ");
	puts(str);
	getch();
}

void str_to_upr(char str[])
{
	char *sptr;
	int i;
	sptr=str;
	for(i=0;*(sptr+i) != '\0';i++)
	{
		if(*(sptr+i)> 96 && *(sptr+i) <123)
		{
			*(sptr+i)-=32;
		}
	}
}
