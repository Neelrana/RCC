#include<stdio.h>
#include<conio.h>
#include<string.h>

void reverse(char[]);
void main()
{
	char str[30];
	printf("Enter string: ");
	gets(str);
	reverse(str);
	printf("String after reverse: ");
	puts(str);
	getch();
}

void reverse(char str[])
{
	char *sptr,temp;
	int length,i;

	sptr=str;
	length=strlen(sptr);

	for(i=0;i<(length/2);i++)
	{
		temp=*(sptr+i);
		*(sptr+i)=*(sptr+length-1-i);
		*(sptr+length-1-i)=temp;
	}

}
