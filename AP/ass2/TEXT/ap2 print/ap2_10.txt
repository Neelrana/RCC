Q(10):	Write a function using pointers to exchange the value stored in two locations in the memory.

*****************************************************************************************************************************************
#include<stdio.h>
#include<conio.h>

void exchange(int*,int*);

void main()
{
	int num1,num2;
	printf("Enter number 1: ");
	scanf("%d",&num1);
	printf("Enter number 2: ");
	scanf("%d",&num2);
	exchange(&num1,&num2);
	printf("After exchange\nNumber 1 is: %d\nnumber 2 is: %d",num1,num2);
	getch();
}

void exchange(int *iptr1,int *iptr2)
{
	int temp;
	temp=*iptr1;
	*iptr1=*iptr2;
	*iptr2=temp;
}
